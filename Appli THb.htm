<script>

const DIVISIONS=['Division 1','Division 2','Division 3'],PHASES=['Phase 1','Phase 2'],POULES=['Poule A','Poule B','Poule C'];

let state={teams:[],matches:[]};

function saveState(){localStorage.setItem('THB_STATE',JSON.stringify(state));}

function loadState(){const s=localStorage.getItem('THB_STATE');if(s)state=JSON.parse(s);}

 

function populateSelectors(){

document.querySelector("#select-division").innerHTML=DIVISIONS.map(d=>`<option>${d}</option>`).join('');

document.querySelector("#select-phase").innerHTML=PHASES.map(p=>`<option>${p}</option>`).join('');

document.querySelector("#select-poule").innerHTML=POULES.map(p=>`<option>${p}</option>`).join('');

document.querySelector("#team-division").innerHTML=document.querySelector("#select-division").innerHTML;

document.querySelector("#team-phase").innerHTML=document.querySelector("#select-phase").innerHTML;

document.querySelector("#team-poule").innerHTML=document.querySelector("#select-poule").innerHTML;

}

 

function updateSelectionDisplay(){

document.querySelector("#sel-div").textContent=document.querySelector("#select-division").value;

document.querySelector("#sel-phase").textContent=document.querySelector("#select-phase").value;

document.querySelector("#sel-poule").textContent=document.querySelector("#select-poule").value;

}

 

function getCurrentTeams(){

const d=document.querySelector("#select-division").value,p=document.querySelector("#select-phase").value,o=document.querySelector("#select-poule").value;

return state.teams.filter(t=>t.division===d&&t.phase===p&&t.poule===o);

}

 

function getCurrentMatches(){

const d=document.querySelector("#select-division").value,p=document.querySelector("#select-phase").value,o=document.querySelector("#select-poule").value;

return state.matches.filter(m=>m.division===d&&m.phase===p&&m.poule===o);

}

 

function updateTeams(){

const teams=getCurrentTeams();

const home=document.querySelector("#match-home"),away=document.querySelector("#match-away");

home.innerHTML=teams.map(t=>`<option>${t.name}</option>`).join('');

away.innerHTML=home.innerHTML;

}

 

function updateRanking(){

const tbody=document.querySelector("#rankings tbody");

const teams=getCurrentTeams();

if(teams.length===0){document.querySelector("#ranking-empty").classList.remove("hidden");tbody.innerHTML="";return;}

document.querySelector("#ranking-empty").classList.add("hidden");

const ranking=teams.map(team=>{

const played=getCurrentMatches().filter(m=>m.homeTeam===team.name||m.awayTeam===team.name);

let v=0,n=0,d=0,gf=0,ga=0,pts=0,form=[];

played.forEach(m=>{

if(m.scoreHome==null)return;

const home=m.homeTeam===team.name;

const goalsFor=home?m.scoreHome:m.scoreAway;

const goalsAgainst=home?m.scoreAway:m.scoreHome;

gf+=goalsFor;ga+=goalsAgainst;

if(goalsFor>goalsAgainst){v++;pts+=3;form.push("🟢");}

else if(goalsFor===goalsAgainst){n++;pts+=2;form.push("🟡");}

else{d++;pts+=1;form.push("🔴");}

});

return {name:team.name,v,n,d,pts,diff:gf-ga,form:form.slice(-5)};

}).sort((a,b)=>b.pts-a.pts||b.diff-a.diff);

tbody.innerHTML=ranking.map((t,i)=>`<tr><td>${i+1}</td><td>${t.name}</td><td>${t.v+t.n+t.d}</td><td>${t.v}</td><td>${t.n}</td><td>${t.d}</td><td>${t.diff}</td><td>-</td><td>${t.pts}</td><td>${t.form.join('')}</td></tr>`).join('');

}

 

function updateMatches(){

const list=document.querySelector("#results-list");

const matches=getCurrentMatches();

if(matches.length===0){document.querySelector("#results-empty").classList.remove("hidden");list.innerHTML="";return;}

document.querySelector("#results-empty").classList.add("hidden");

list.innerHTML=matches.map(m=>`<div class="match-card ${m.scoreHome==null?'match-pending':''}">${m.date} : ${m.homeTeam} ${m.scoreHome??'-'} - ${m.scoreAway??'-'} ${m.awayTeam}</div>`).join('');

}

 

function handleAddTeam(){

const name=document.querySelector("#team-name").value.trim();

const d=document.querySelector("#team-division").value,p=document.querySelector("#team-phase").value,o=document.querySelector("#team-poule").value;

if(!name)return alert("Nom requis");

state.teams.push({name,division:d,phase:p,poule:o});

saveState();updateTeams();updateRanking();

}

 

function handleAddMatch(){

const date=document.querySelector("#match-date").value;

const home=document.querySelector("#match-home").value,away=document.querySelector("#match-away").value;

const sh=parseInt(document.querySelector("#match-score-home").value),sa=parseInt(document.querySelector("#match-score-away").value);

const d=document.querySelector("#select-division").value,p=document.querySelector("#select-phase").value,o=document.querySelector("#select-poule").value;

if(home===away)return alert("Deux quipes identiques");

if(state.matches.some(m=>m.date===date&&m.homeTeam===home&&m.awayTeam===away&&m.division===d&&m.phase===p&&m.poule===o))return alert("Match dj existant !");

state.matches.push({id:Date.now(),date,division:d,phase:p,poule:o,homeTeam:home,awayTeam:away,scoreHome:isNaN(sh)?null:sh,scoreAway:isNaN(sa)?null:sa});

saveState();updateRanking();updateMatches();

}

 

// Navigation

document.querySelectorAll(".tab").forEach(tab=>{

tab.addEventListener("click",()=>{

document.querySelectorAll(".tab").forEach(t=>t.classList.remove("active"));

tab.classList.add("active");

document.querySelectorAll("main section").forEach(sec=>sec.classList.add("hidden"));

document.querySelector(`#${tab.dataset.tab}`).classList.remove("hidden");

});

});

 

document.querySelector("#select-division").addEventListener('change',()=>{updateSelectionDisplay();updateTeams();updateRanking();updateMatches();});

document.querySelector("#select-phase").addEventListener('change',()=>{updateSelectionDisplay();updateTeams();updateRanking();updateMatches();});

document.querySelector("#select-poule").addEventListener('change',()=>{updateSelectionDisplay();updateTeams();updateRanking();updateMatches();});

 

// Initialisation

loadState();populateSelectors();updateSelectionDisplay();updateTeams();updateRanking();updateMatches();

</script>
